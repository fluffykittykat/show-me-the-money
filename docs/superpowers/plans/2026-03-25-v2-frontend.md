# V2 Frontend: Story-First Redesign

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the existing multi-fetch, tabbed frontend with a story-first, vertical-scroll design powered by single v2 API calls per page.

**Architecture:** In-place rewrite of existing Next.js App Router pages. Add v2 types and API functions to existing `lib/` files. Build new components for verdicts, money trails, and story cards. Replace page components one at a time. No new dependencies.

**Tech Stack:** Next.js 16 (App Router), React 19, TypeScript 5.9, Tailwind CSS v4, Lucide React

**Spec:** `docs/superpowers/specs/2026-03-25-v2-frontend-design.md`

**IMPORTANT:** This is NOT the Next.js you know. Before writing any code, read the relevant guide in `frontend/node_modules/next/dist/docs/` to check for breaking changes in APIs, conventions, and file structure.

```tsx
'use client';

import { useState } from 'react';
import { RefreshCw, Zap } from 'lucide-react';
import { getBriefing } from '@/lib/api';

interface AIBriefingProps {
  briefing: string | null;
  slug: string;
  className?: string;
}

export default function AIBriefing({ briefing, slug, className = '' }: AIBriefingProps) {
  const [text, setText] = useState(briefing);
  const [loading, setLoading] = useState(false);

  async function handleRegenerate() {
    setLoading(true);
    try {
      const res = await getBriefing(slug, true);
      setText(res.briefing);
    } catch {
      // silently fail — keep existing text
    } finally {
      setLoading(false);
    }
  }

  return (
    <div className={`rounded-xl border border-amber-500/20 p-5 mb-8 ${className}`}
         style={{ background: 'linear-gradient(135deg, rgba(245,158,11,0.08), rgba(245,158,11,0.02))' }}>
      <div className="flex items-center gap-2 mb-3 text-amber-400 font-semibold text-sm uppercase tracking-wide">
        <Zap className="w-4 h-4" />
        AI Assessment
      </div>
      {text ? (
        <p className="text-zinc-300 leading-relaxed text-[0.95rem]">{text}</p>
      ) : (
        <p className="text-zinc-500 italic">No briefing generated yet.</p>
      )}
      <button
        onClick={handleRegenerate}
        disabled={loading}
        className="mt-3 inline-flex items-center gap-1.5 px-3 py-1 rounded-md border border-amber-500/30 text-amber-400 text-xs hover:bg-amber-500/10 disabled:opacity-50 transition-colors"

        {loading ? 'Generating...' : text ? 'Regenerate' : 'Generate briefing'}
      </button>
    </div>
  );
}
```

```tsx
'use client';

import Link from 'next/link';
import { formatMoney } from '@/lib/utils';

interface ChainNode {
  label: string;
  name: string;
  slug: string;
  href: string;
  amount?: number;
  amountLabel?: string;
}

interface ChainVisualizationProps {
  chain: {
    donors?: Array<{ name: string; slug: string; amount: number }>;
    committees?: Array<{ name: string; slug: string }>;
    bills?: Array<{ name: string; slug: string }>;
    middlemen?: Array<{ name: string; slug: string; amount_in: number; amount_out: number }>;
    lobbying?: Array<{ firm: string; client: string; issue: string }>;
  };
  officialName: string;
  officialSlug: string;
  className?: string;
}

export default function ChainVisualization({ chain, officialName, officialSlug, className = '' }: ChainVisualizationProps) {
  const nodes: ChainNode[] = [];

  // Donors
  const donors = chain.donors || [];
  if (donors.length > 0) {
    const d = donors[0];
    nodes.push({
      label: 'Donor',
      name: d.name,
      slug: d.slug,
      href: `/entities/pac/${d.slug}`,
      amount: d.amount,
    });
  }

  // Middlemen
  const middlemen = chain.middlemen || [];
  if (middlemen.length > 0) {
    const m = middlemen[0];
    nodes.push({
      label: 'Middleman',
      name: m.name,
      slug: m.slug,
      href: `/entities/pac/${m.slug}`,
      amountLabel: `${formatMoney(m.amount_in)} in → ${formatMoney(m.amount_out)} out`,
    });
  }

  // Official
  nodes.push({
    label: 'Official',
    name: officialName,
    slug: officialSlug,
    href: `/officials/${officialSlug}`,
  });

  // Committees
  const committees = chain.committees || [];
  if (committees.length > 0) {
    const c = committees[0];
    nodes.push({
      label: 'Committee',
      name: c.name,
      slug: c.slug,
      href: `/entities/organization/${c.slug}`,
    });
  }

  // Bills
  const bills = chain.bills || [];
  if (bills.length > 0) {
    const b = bills[0];
    nodes.push({
      label: 'Legislation',
      name: b.name,
      slug: b.slug,
      href: `/bills/${b.slug}`,
    });
  }

  if (nodes.length === 0) return null;

  return (
    <div className={`flex items-center gap-0 overflow-x-auto py-2 flex-wrap ${className}`}>
      {nodes.map((node, i) => (
        <div key={i} className="flex items-center">
          {i > 0 && <span className="text-zinc-500 text-lg px-1.5 flex-shrink-0">→</span>}
          <div className="bg-zinc-800 border border-zinc-700 rounded-lg px-3 py-2 text-center min-w-0">
            <span className="block text-[0.65rem] uppercase tracking-wide text-zinc-500">{node.label}</span>
            <Link href={node.href} className="block text-sm font-medium text-zinc-100 hover:text-amber-400 transition-colors truncate">
              {node.name}
            </Link>
            {node.amount != null && (
              <span className="block text-xs text-amber-400">{formatMoney(node.amount)}</span>
            )}
            {node.amountLabel && (
              <span className="block text-xs text-amber-400">{node.amountLabel}</span>
            )}
          </div>
        </div>
      ))}
    </div>
  );
}
```
